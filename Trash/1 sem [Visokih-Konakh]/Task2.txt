#include<iostream>
#include<math.h>
using namespace std;
int main()
{
	setlocale(LC_ALL, ".1251");
	int a, i,c=0, MAX=0, t ,MaxC=0,n=0,MACC=0,g=0;
	cin >> t;
	do
	{
		for (i = 0; i < 10; i++)
		{
			a = t;
			while (a > 0)
			{
				if (a % 10 == i)
					c++;
				a /= 10;
			}
			if (c > MaxC)
			{
				MaxC = c;
				MAX = i;
			}
			c = 0;
		}
		if (g > 0)
		{
			if (MaxC == MACC)
				cout << "Внимвние! Найдена другая цифра, повторяющаяся максимальное количество раз = " << MAX << endl;
			else
			{
				cout << "Число с удалённой цифрой(-ами) = " << t << endl;
				system("pause");
				return 0;
			}
		}
		a = 0;
		while (t > 0)
		{
			if ((t % 10) != MAX)
			{
				a = (t % 10) * pow(10, n) + a;
				n++;
			}
			t /= 10;
		}
		n = 0;
		if(g==0)
		cout << "Цифра повторяющаяся максимальное количество раз = " << MAX << endl;
		cout << "Количество повторений максимальной цифры = " << MaxC << endl;
		MACC = MaxC;
		MaxC = 0;
		MAX = 0;
		t = a;
		i = 0;
		g++;
	} while (true);
	
}
